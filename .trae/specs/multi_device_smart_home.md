### 场景2：分布式媒体播放

#### 2.1 音乐跟随

```
你在厨房做饭
    ↓
对手机A说："播放周杰伦的歌"
    ↓
厨房手机A开始播放
    ↓
你走到厕所
    ↓
厕所手机B自动检测到人（距离传感器）
    ↓
音乐自动切换到厕所手机B播放
    ↓
你走到卧室
    ↓
卧室手机C自动检测到人
    ↓
音乐自动切换到卧室手机C播放
```

**实现逻辑**:
```python
class MediaFollower:
    def __init__(self):
        self.current_phone = "kitchen"
        self.user_location = "kitchen"
        
    def on_proximity_change(self, phone_id: str, detected: bool):
        if detected:
            old_location = self.user_location
            self.user_location = phone_id
            
            # 检测到新人位置，切换播放
            if old_location != phone_id:
                self.transfer_media(old_location, phone_id)
    
    def transfer_media(self, from_phone: str, to_phone: str):
        # 获取当前播放状态
        status = self.get_playback_status(from_phone)
        
        # 在新设备恢复播放
        self.set_playback_status(to_phone, status)
        
        # 暂停旧设备
        self.pause_playback(from_phone)
        
        print(f"音乐从 {from_phone} 切换到 {to_phone}")
```

```python
    "from": "kitchen_phone",

        "track_id": "uuid",
        "position_ms": 123456,

        "model": "qwen2.5-3b",
        "prompt": "番茄炒蛋怎么做？",
        "multimodal": {...}
    }
}
```

### 数据同步

```python
# PC端 - 多设备数据融合
class MultiDeviceFusion:
    def __init__(self):
        self.devices = {
            "kitchen": DeviceState(),
            "bathroom": DeviceState(),
            "bedroom": DeviceState(),
            "mobile": DeviceState(),
        }
        
    def on_sensor_data(self, device_id: str, data: dict):
        # 更新设备状态
        self.devices[device_id].update(data)
        
        # 融合所有传感器数据
        fused = self.fuse_all_sensors()
        
        # 智能决策
        decisions = self.make_decisions(fused)
        
        # 执行决策
        for action in decisions:
            self.execute(action)
    
    def fuse_all_sensors(self) -> dict:
        # 计算全屋平均温度
        avg_temp = np.mean([
            d.sensors.get("temp", 0) 
            for d in self.devices.values()
        ])
        
        # 检测是否有人
        occupancy = {
            device_id: d.sensors.get("distance", 999) < 100
            for device_id, d in self.devices.items()
        }
        
        return {
            "avg_temp": avg_temp,
            "occupancy": occupancy,
            # ...
        }
```

## 功耗管理

### 固定设备（厨房/厕所/卧室）

```python
# 低功耗模式
class PowerSaving:
    def __init__(self):
        self.idle_timeout = 300  # 5分钟
        
    def enter_low_power(self):
        # 关闭屏幕
        self.screen.off()
        
        # 降低传感器采样率
        self.sensor_rate = 1  # 1Hz
        
        # 关闭WiFi，用蓝牙
        self.wifi.off()
        self.bluetooth.on()
        
        # 功耗：< 0.5W
```
